  <script>
    function calculateClientsNeeded({ incomeGoal, averageServicePrice, frequencyPerClientPerMonth, retentionRate, monthsPerYear = 12 }) {
      if (
        incomeGoal <= 0 ||
        averageServicePrice <= 0 ||
        frequencyPerClientPerMonth <= 0 ||
        monthsPerYear <= 0 ||
        retentionRate <= 0 || retentionRate > 1
      ) {
        return 'All input values must be greater than 0, and retention rate must be between 0 and 1.';
      }


      const totalSessionsNeeded = incomeGoal / averageServicePrice;
      const sessionsPerClientPerYear = frequencyPerClientPerMonth * monthsPerYear;
      const adjustedClientsNeeded = totalSessionsNeeded / (sessionsPerClientPerYear * retentionRate);
      return Math.ceil(adjustedClientsNeeded);
    }


    document.getElementById('calculatorForm').addEventListener('submit', function(e) {
      e.preventDefault();
      const incomeGoal = parseFloat(document.getElementById('incomeGoal').value);
      const avgPrice = parseFloat(document.getElementById('avgPrice').value);
      const visits = parseFloat(document.getElementById('visitsPerMonth').value);
      const retention = parseFloat(document.getElementById('retentionRate').value);
      const errorMsg = document.getElementById('errorMsg');
      const resultDiv = document.getElementById('result');


      errorMsg.textContent = '';
      resultDiv.textContent = '';


      const result = calculateClientsNeeded({
        incomeGoal: incomeGoal,
        averageServicePrice: avgPrice,
        frequencyPerClientPerMonth: visits,
        retentionRate: retention
      });


      if (typeof result === 'string') {
        errorMsg.textContent = result;
      } else {
        resultDiv.textContent = `You need approximately ${result} active clients to meet your income goal.`;
      }
    });
  </script>
